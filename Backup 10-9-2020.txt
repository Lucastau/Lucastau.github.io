<script>
    var sel = document.getElementById('champOptions');
    let champData;
    $(document).ready(function () {
        $('.js-example-basic-single').select2();


        $('.js-example-basic-single').on('change', function (e) {

            let selectedChamp = champData.data[sel.options[sel.selectedIndex].value]

            console.log(selectedChamp.version);
            console.log(selectedChamp.info);

            document.getElementById("Patch").innerHTML = JSON.stringify(selectedChamp.version)
            document.getElementById("Stats").innerHTML = JSON.stringify(selectedChamp.info)
            document.getElementById("Class").innerHTML = JSON.stringify(selectedChamp.tags)
        });
    });
    fetch(
        `https://ddragon.leagueoflegends.com/cdn/10.20.1/data/en_US/champion.json`
    ).then(response => response.json())
        .then(stuff => {
            champData = stuff;

            for (const lol of Object.keys(stuff.data)) {

                // create new option element
                var opt = document.createElement('option');
                // create text node to add to option element (opt)
                opt.appendChild(document.createTextNode(stuff.data[lol].name));

                // set value property of opt
                opt.value = lol;

                // add opt to end of select box (sel)
                sel.appendChild(opt);

            }

        });

</script>
